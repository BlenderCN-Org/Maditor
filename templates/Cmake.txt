cmake_minimum_required (VERSION 2.6)

cmake_policy(SET CMP0011 OLD)
cmake_policy(SET CMP0049 OLD)

%5

set (sources %3)

if (NOT sources STREQUAL "")  
    add_library(%2 ${MADGINE_LIB_TYPE} ${sources})
endif()

%4

target_link_libraries (%2 ${MADGINE_LIBRARIES})

set_target_properties(%2 PROPERTIES LINKER_LANGUAGE CXX)
