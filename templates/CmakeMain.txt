cmake_minimum_required (VERSION 2.6)

cmake_policy(SET CMP0011 OLD)
cmake_policy(SET CMP0049 OLD)


PROJECT(%1)

include(Dependencies.cmake)

function(include_library lib)

    find_package(${lib})
                        
    foreach (target ${ARGN})

        get_target_property(dirs ${target} INTERFACE_INCLUDE_DIRECTORIES)
        include_directories(${dirs})

    endforeach()

endfunction(include_library)

include_library(Interfaces Interfaces)
include_library(Madgine OgreMadgine MadgineServer)

set(INTERFACES_LIBRARIES Interfaces)
set(MADGINE_LIBRARIES OgreMadgine MadgineServer)

%2

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ${CMAKE_SOURCE_DIR}/../debug/bin)

set(CMAKE_MODULE_PATH
	"${CMAKE_SOURCE_DIR}/../CMake"
)

include_directories(${CMAKE_CURRENT_SOURCE_DIR})

%3
